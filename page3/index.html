    2022-08-16
    Takuo Kihira
    JavaScript
    Tweet

    先日、次のような JavaScript クイズを Twitter で出しました。

    // JavaScript quiz: 出力は？
    const a = [];
    {
    for(let i = 0; i < 10; i++) {
    a[i] = () => console.log(i);
    }
    }
    a[3]();
    {
    let i;
    for(i = 0; i < 10; i++) {
    a[i] = () => console.log(i);
    }
    }
    a[3]();
    {
    for(let i = 0; i < 10;) {
    a[i] = () => console.log(i);
    i++;
    }
    }
    a[3]();
